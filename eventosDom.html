EVENTOS DEL MOUSE
EVENTO DESCRIPCION EJEMPLO
Click cuando se pulsa el botón izquierdo del ratón // seleccionar le nodo donde se nesecita un evento
const button = document.querySelector('#button');
// Crear la Funcion deseada segun el evento
button.addEventListener('click', () => { console.log('Me HIciste Click') })
dblclick cuando se pulsa dos veces seguidas el botón izquierdo del ratón // seleccionar le nodo donde se nesecita un evento
const button = document.querySelector('#button');
// Crear la Funcion deseada segun el evento
button.addEventListener('dblclick',()=> { console.log('DOBLE CLICK') });
mouseenter cuando se entra en la zona que tiene el evento // seleccionar le nodo donde se nesecita un evento
const button = document.querySelector('#box');
// 2. Seleccionar elemento hijo a reeemplazar
box.addEventListener('mouseenter', () => { box.classList.replace('red', 'green') })
